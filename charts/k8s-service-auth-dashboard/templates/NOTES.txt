Thank you for installing {{ .Chart.Name }}!

Your release is named {{ .Release.Name }}.

To get the admin credentials:

  export ADMIN_USER=$(kubectl get secret -n {{ .Release.Namespace }} {{ include "k8s-service-auth-dashboard.fullname" . }}-admin-user -o jsonpath='{.data.username}' | base64 -d)
  export ADMIN_PASS=$(kubectl get secret -n {{ .Release.Namespace }} {{ include "k8s-service-auth-dashboard.fullname" . }}-admin-user -o jsonpath='{.data.password}' | base64 -d)
  echo "Username: $ADMIN_USER"
  echo "Password: $ADMIN_PASS"

To access the dashboard:

  {{- if .Values.frontend.ingress.enabled }}
  {{- range $host := .Values.frontend.ingress.hosts }}
  http://{{ $host.host }}
  {{- end }}
  {{- else }}
  export POD_NAME=$(kubectl get pods -n {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "k8s-service-auth-dashboard.name" . }},app.kubernetes.io/component=frontend" -o jsonpath="{.items[0].metadata.name}")
  kubectl port-forward -n {{ .Release.Namespace }} $POD_NAME 3000:3000
  Then open http://localhost:3000
  {{- end }}

To check the status of the deployment:

  kubectl get pods -n {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "k8s-service-auth-dashboard.name" . }}"
